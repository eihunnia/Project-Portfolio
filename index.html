public void ExpandNode(Node node) {
    foreach (var neighbour in node.neighbours) {
        var cost = node.cost + Distance(node, neighbour);
        if (cost < neighbour.cost) {
            neighbour.cost = cost;
            neighbour.parent = node;
        }
    }
}
